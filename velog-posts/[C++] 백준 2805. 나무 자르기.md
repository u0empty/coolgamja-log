<pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;algorithm&gt;
#include &lt;vector&gt;
using namespace std;

int N, K;
vector&lt;int&gt; v;

long long solution() {
    long long l = 0;
    long long r = v[N - 1];
    long long m, sum, h = 0;
    while (l &lt;= r) {
        m = (l + r) / 2;
        sum = 0;
        for (int i = 0; i &lt; N; i++) {
            if (v[i] &lt;= m) continue;
            sum += v[i] - m;
        }
        if (sum &lt; K) {
            r = m - 1;
        }
        else {
            l = m + 1;
            h = max(h, m);
        }
    }

    return h;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin &gt;&gt; N &gt;&gt; K;
    v.resize(N);
    for (int i = 0; i &lt; N; i++) cin &gt;&gt; v[i];
    sort(v.begin(), v.end());

    cout &lt;&lt; solution();

    return 0;
