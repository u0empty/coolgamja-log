#include &lt;algorithm&gt;

    int n, maxi = 0;
    int dp[500][500] = { 0, };

            else if (i == j) dp[i][j] = dp[i - 1][j - 1] + dp[i][j];
            else dp[i][j] = dp[i][j] + max(dp[i - 1][j - 1], dp[i - 1][j]);
        }
    }

    for (int i = 0; i &lt; n; i++) {
        maxi = max(maxi, dp[n - 1][i]);
    }
    cout &lt;&lt; maxi;
    return 0;
