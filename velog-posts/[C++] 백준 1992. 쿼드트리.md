void getTree(int y, int x, int s) {
    char c = img[y][x];
    for (int i = y; i &lt; y + s; i++) {

                tree += ')';
                return;
            }
        }
    }
    tree += c;
}

    ios::sync_with_stdio(0);

    img.resize(n);
