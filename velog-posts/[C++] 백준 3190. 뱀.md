int N, K, L, t = 0;

void getNewRc1(int dr) {
    if (dr == 0) {
        nr = r - 1;
        nc = c;
    }
    else if (dr == 1) {
        nr = r;
        nc = c + 1;
    }
    else if (dr == 2) {
        nr = r + 1;
        nc = c;
    }
    else {
        nr = r;
        nc = c - 1;
    }
}

void solution() {
    r = 1;
    c = 1;

    path.push({ r, c });
    int dr = 1; // 0ìƒ, 1ìš°, 2í•˜, 3ì¢Œ

                else dr == 3 ? dr = 0 : dr += 1;
                q.pop();

        }

        path.push({ nr, nc });
        r = nr;
        c = nc;
    }
}
